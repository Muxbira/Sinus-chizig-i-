# Google Colab uchun sin(x) funksiyasini grafigini chizish
import numpy as np
import matplotlib.pyplot as plt

# X o'qidagi qiymatlarni aniqlash (masalan, -2π dan 2π gacha)
x = np.linspace(-2 * np.pi, 2 * np.pi, 1000)
y = np.sin(x)

# Grafikni chizish
plt.figure(figsize=(10, 5))
plt.plot(x, y, label='sin(x)', color='blue')
plt.title('sin(x) funksiyasining grafigi')
plt.xlabel('x')
plt.ylabel('sin(x)')
plt.axhline(0, color='black', linewidth=0.5, linestyle='--')  # x o'qini chizish
plt.axvline(0, color='black', linewidth=0.5, linestyle='--')  # y o'qini chizish
plt.grid(True)
plt.legend()
plt.show()
